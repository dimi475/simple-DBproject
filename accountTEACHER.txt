 
 /*Create tables*/
 CREATE TABLE COAST_LINE (
 CL_OBJECT_NO INT,
 CL_OBJECT_NAME VARCHAR(45),
 CL_LATITUDE VARCHAR(45),
 CL_LONGITUDE VARCHAR(45),
 CONSTRAINT CL_OBJECT_NO_pk PRIMARY KEY (CL_OBJECT_NO));
 
 CREATE TABLE TRACK_TABLE (
 TRACK_TABLE_NO INT,
 TRACK_POINT_NO INT,
 P_LATITUDE VARCHAR(45),
 P_LONGITUDE VARCHAR(45),
 CONSTRAINT TRACK_TABLE_NO_pk PRIMARY KEY (TRACK_TABLE_NO));
 
 CREATE TABLE TRACK_LINE (
 TRACK_LINE_NO INT,
 TRACK_TABLE_NO INT,
 CONSTRAINT TRACK_LINE_NO_pk PRIMARY KEY (TRACK_LINE_NO),
 CONSTRAINT TRACK_TABLE_NO_fk FOREIGN KEY (TRACK_TABLE_NO) REFERENCES TRACK_TABLE (TRACK_TABLE_NO));
 
 create table CHART (
 TABLE_NUM VARCHAR(15),
 CL_OBJECT_NO INT,
 TRACK_LINE_NO INT,
 CONSTRAINT CHART_table_pk PRIMARY KEY (TABLE_NUM),
 CONSTRAINT CL_OBJECT_NO_fk FOREIGN KEY (CL_OBJECT_NO) REFERENCES COAST_LINE(CL_OBJECT_NO),
 CONSTRAINT TRACK_LINE_NO_fk FOREIGN KEY (TRACK_LINE_NO) REFERENCES TRACK_LINE(TRACK_LINE_NO));
